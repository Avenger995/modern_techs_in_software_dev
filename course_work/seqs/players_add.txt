@startuml
'seq diag for players (add)


actor User as user
participant "UI" as ui
participant "API" as api
participant "SQL" as sql

user -> ui: Добавление нового игрока
ui --> user: Модульное окно
user -> ui: Ввод и отправка параметров игрока
activate ui
ui -> api: POST /api/players
activate api
api -> api: Маппинг DTO к модели 
api -> sql: Отправка модели
activate sql
sql --> api: Модель успешно добавлена
deactivate sql
api --> ui: Статус 200
deactivate api
ui --> user: Высплвающее окно об успешности
deactivate ui

@enduml
